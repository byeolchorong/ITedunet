// 좌우 슬라이드
let left = document.querySelector(".left");
let right = document.querySelector(".right");
let slide = document.querySelectorAll(".slide");

left.addEventListener("click",leftsliding);
right.addEventListener("click",rightsliding);

let index = 0;

function leftsliding() {
  index--;

  if (index < 0) {
    index = slide.length -1;
  }
  sliding();
}

function rightsliding() {
  index++;

  if (index >= slide.length) {
    index = 0;
  }
  sliding();
}

function sliding() {
  slide.forEach(function(slide) {
    slide.style.transform = `translateX(${index*(-100)}%)`;
  });
}


// 도트 슬라이드
let btn = document.querySelectorAll("label");
let slides = document.querySelectorAll(".slide");

for (let i = 0; i < btn.length; i++) {
  btn[i].addEventListener("click",slidings);
}

let indexs = 0;

function slidings() {
  indexs = parseInt (this.id.charAt(this.id.length-1));

  slides.forEach(function(slides) {
    slides.style.transform = `translate(${indexs * (-100)}%)`;
  })
}
