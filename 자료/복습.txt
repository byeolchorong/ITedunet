function removeItem() {
	// removeChild(), this
	// remove() : 본인을 삭제하겠다.
	// this : 이벤트가 발생한 객체 : span태그(X)
	// 방법1
	let pNode = this.parentNode.parentNode;
	let removeNode = this.parentNode;
	
	pNode.removeChild(removeNode);
	
	// 방법2
	// 클릭한 요소의 li태그
	this.parentNode.remove();
}
